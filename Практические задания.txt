Задание 1: Настройка FTP с анонимным доступом.
apt-get update && apt-get install -y vsftpd
systemctl status vsftpd
systemctl enable -–now vsftpd
mkdir -p /var/ftp/uploads
chmod 777 /var/ftp/uploads
chown ftp:ftp /var/ftp/uploads
cat > /etc/vsftpd.conf << EOF
# Разрешить анонимный доступ
anonymous_enable=YES
# Каталог для анонимных пользователей
anon_root=/var/ftp
# Разрешить загрузку файлов анонимными пользователями
write_enable=YES
anon_upload_enable=YES
anon_mkdir_write_enable=YES
anon_other_write_enable=YES
EOF
systemctl restart vsftpd
systemctl status vsftpd
Проверьте подключение с другого компьютера:
ftp <IP-Адрес сервера>

Задание 3: Настройка NAT для доступа внутренних узлов в интернет.
hostnamectl set-hostname isp.au-team.irpo:; exec bash
mkdir /etc/net/ifaces/ens19 /etc/net/ifaces/ens20
echo "172.16.1.1/28" > /etc/net/ifaces/ens19/ipv4address
echo "172.16.2.1/28" > /etc/net/ifaces/ens20/ipv4address
cat > /etc/net/ifaces/ens19/options << EOF 
BOOTPROTO=static
TYPE=eth
NM_CONTROLLED=no
DISABLED=no
CONFIG_WIRELESS=no
SYSTEMD_BOOTPROTO=static
CONFIG_IPV4=yes
SYSTEMD_CONTROLLED=no
EOF
cat > /etc/net/ifaces/ens20/options << EOF 
BOOTPROTO=static
TYPE=eth
NM_CONTROLLED=no
DISABLED=no
CONFIG_WIRELESS=no
SYSTEMD_BOOTPROTO=static
CONFIG_IPV4=yes
SYSTEMD_CONTROLLED=no
EOF
systemctl restart network
sed -i 's/net.ipv4.ip_forward = 0/net.ipv4.ip_forward = 1/' /etc/net/sysctl.conf
apt-get update && apt-get install -y iptables tzdata
iptables -t nat -A POSTROUTING -o ens18 -j MASQUERADE -s 172.16.1.0/28
iptables -t nat -A POSTROUTING -o ens18 -j MASQUERADE -s 172.16.2.0/28
iptables -A FORWARD -i ens19 -o ens18 -s 172.16.1.0/28 -j ACCEPT
iptables -A FORWARD -i ens20 -o ens18 -s 172.16.2.0/28 -j ACCEPT
iptables-save > /etc/sysconfig/iptables
sysctl -p /etc/sysctl.conf
systemctl enable --now iptables
systemctl restart iptables
timedatectl set-timezone Asia/Yekaterinburg

Задание 4: Создание SMB-ресурса.
apt-get update && apt-get install -y samba
mkdir /home/docstore
mv /etc/samba/smb.conf /etc/samba/smb.conf.old
cat > /etc/samba/smb.conf << EOF
[global]
dos charset = CP866
unix charset = utf8
#имя рабочей группы
workgroup = WORKGROUP
#имя сервера
server string = Filestore
#группа пользователей
security = USER
map to guest = Bad User

#имя ресурса
[Public] 
#путь к папке
path = /home/docstore 
guest ok = Yes
browseable = yes
writable = yes
create mask = 0777
directory mask = 0777
EOF
mkdir -p /var/lib/samba/private
systemctl restart smb.service nmb.service
systemctl enable --now smb.service nmb.service

Задание 6: Развёртывание Nginx.

apt-get update && apt-get install nginx -y
cat > /etc/nginx/sites-available.d/default.conf << EOF
server {
        listen 80;
        server_name localhost .local 192.168.1.26;   

	access_log /var/log/nginx/access.log;
	error_log /var/log/nginx/error.log;

        location / {
		root /var/www/html/;
		autoindex on;
        }
}
EOF
ln -s /etc/nginx/sites-available.d/default.conf /etc/nginx/sites-enabled.d/

mkdir /var/www/html/
touch /var/www/html/index.html
cat > /var/www/html/index.html << EOF
<html><body><h1>It works! Nginx</h1></body></html>
EOF
systemctl enable --now nginx
systemctl restart nginx
systemctl status nginx

Сайт будет доступен по адресам:
http://IP-сервера
