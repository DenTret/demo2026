# HQ-RTR
```bash
en
conf
hostname hq-rtr.au-team.irpo
end
wr mem

en
conf
int HQ-RTR-ISP
ip address 172.16.1.2/28
port te0
service-inst 0
enc untagged
conn ip int HQ-RTR-ISP
exit

int HQ-RTR-SRV
ip address 192.168.100.1/27
port te1
service-inst 100
enc dot1q 100
rewrite pop 1
conn ip int HQ-RTR-SRV
exit

int HQ-RTR-CLI
ip address 192.168.200.1/27
port te1
service-inst 200
enc dot1q 200
rewrite pop 1
conn ip int HQ-RTR-CLI
exit

int VLAN999
ip address 192.168.99.1/29
port te1
service-inst 999
enc dot1q 999
rewrite pop 1
conn ip int VLAN999
end
wr mem

en
conf
ip route 0.0.0.0 0.0.0.0 172.16.1.1
end
wr mem

en
conf
username net_admin
password P@ssw0rd
role admin
end
wr mem

en
conf
int tunnel.0
ip address 172.16.0.1/30
ip mtu 1400
ip ospf network broadcast   
ip ospf mtu-ignore   
ip tunnel 172.16.1.2 172.16.2.2 mode gre
end
wr mem

en
conf
router ospf 1
router-id 1.1.1.1
network 172.16.0.0/30 area 0
network 192.168.100.0/27 area 0
network 192.168.200.0/27 area 0
passive-interface default
no passive-interface tunnel.0
end
wr mem

en
conf
router ospf 1  
area 0 authentication  
ex  
interface tunnel.0   
ip ospf authentication-key ecorouter  
end
wr mem   

en
conf
int HQ-RTR-ISP
ip nat outside
int HQ-RTR-SRV
ip nat inside
int HQ-RTR-CLI
ip nat inside
int VLAN999
ip nat inside
exit
ip nat pool HQ 192.168.100.1-192.168.100.30,192.168.200.1-192.168.200.30
ip nat source dynamic inside-to-outside pool HQ overload interface HQ-RTR-ISP
end
wr mem

en
conf
ip pool HQ-CLI 192.168.200.2-192.168.200.30
dhcp-server 1
pool HQ-CLI 1
mask 27
gateway 192.168.200.1
dns 192.168.100.30
domain-name au-team.irpo
exit
interface HQ-RTR-CLI
dhcp-server 1
end
wr mem

en
conf
ntp timezone utc+5
do show ntp timezone
end
wr mem
```
# HQ-SRV
```bash
hostnamectl set-hostname hq-srv.au-team.irpo; exec bash

cat > /etc/net/ifaces/ens18/options << EOF 
BOOTPROTO=static
TYPE=eth
CONFIG_WIRELESS=no
SYSTEMD_BOOTPROTO=static
CONFIG_IPV4=yes
DISABLED=no
NM_CONTROLLED=no
SYSTEMD_CONTROLLED=no
EOF

echo "192.168.100.30/27" > /etc/net/ifaces/ens18/ipv4address

echo "default via 192.168.100.1" > /etc/net/ifaces/ens18/ipv4route

systemctl restart network
```
```bash
useradd -u 2026 -G wheel sshuser
passwd sshuser
```
```bash
echo "sshuser ALL=(ALL) NOPASSWD: ALL" >> /etc/sudoers

cat >> /etc/openssh/sshd_config << EOF
Port 2026
MaxAuthTries 2
PasswordAuthentication yes
AllowUsers    sshuser
Banner /etc/openssh/bannermotd
EOF

echo "Authorized access only" > /etc/openssh/bannermotd

systemctl restart sshd

echo "nameserver 8.8.8.8" >> /etc/resolv.conf

systemctl restart network

apt-get update && apt-get install -y dnsmasq 
```
```bash
cat > /etc/dnsmasq.conf << EOF
listen-address=192.168.100.30
domain=au-team.irpo
strict-order
no-resolv
server=8.8.8.8
server=8.8.4.4
address=/isp.au-team.irpo/192.168.1.25 #Изменить на ip-адрес порта ens18 у isp
address=/hq-rtr.au-team.irpo/192.168.100.1
address=/hq-srv.au-team.irpo/192.168.100.30

address=/br-rtr.au-team.irpo/192.168.0.1

EOF
```

```bash
systemctl restart dnsmasq
```
